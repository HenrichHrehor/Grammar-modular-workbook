  <script>
    // Function to update the displayed date on all sheets
    function updateDateDisplay() {
      const dateInput = document.querySelector('.control-panel .date-input');
      const dateDisplays = document.querySelectorAll('.sheet .date-display');
      dateDisplays.forEach(span => {
        span.textContent = dateInput.value;
      });
    }

    // Function to print a specific page
    function printSelectedPage(pageNum) {
      const printableArea = document.querySelector('.printable-area');
      const allSheets = printableArea.querySelectorAll('.sheet');
      const controlPanel = document.querySelector('.control-panel');

      // Update date before printing
      updateDateDisplay();

      // Temporarily hide all sheets
      allSheets.forEach(sheet => {
        sheet.style.display = 'none';
      });

      // Show only the selected sheet
      const targetSheet = document.getElementById(`page-content-${pageNum}`);
      if (targetSheet) {
        targetSheet.style.display = 'flex'; // Use flex as it's the default for .sheet
      } else {
        console.error(`Page content for page ${pageNum} not found.`);
        return;
      }

      // Hide control panel during print
      if (controlPanel) {
        controlPanel.style.display = 'none';
      }

      window.print();

      // Restore original display styles
      if (controlPanel) {
        controlPanel.style.display = 'flex'; // Restore flex display for control-panel
      }
      allSheets.forEach(sheet => {
        sheet.style.display = 'flex'; // Restore flex display for sheets
      });
    }

    // Function to print all pages
    function printAllPages() {
      const printableArea = document.querySelector('.printable-area');
      const allSheets = printableArea.querySelectorAll('.sheet');
      const controlPanel = document.querySelector('.control-panel');

      // Update date before printing
      updateDateDisplay();

      // Ensure all sheets are visible for "print all"
      allSheets.forEach(sheet => {
        sheet.style.display = 'flex'; // Make sure all are visible with flex
      });

      // Hide control panel during print
      if (controlPanel) {
        controlPanel.style.display = 'none';
      }

      window.print();

      // Restore control panel
      if (controlPanel) {
        controlPanel.style.display = 'flex';
      }
    }

    // Initialize date and event listeners when the DOM is ready
    document.addEventListener('DOMContentLoaded', () => {
      const dateInput = document.querySelector('.control-panel .date-input');
      // Set initial date to today for convenience
      const today = new Date();
      dateInput.value = today.toISOString().split('T')[0];
      updateDateDisplay(); // Update display immediately after setting initial value

      // Listen for changes in the date input
      dateInput.addEventListener('change', updateDateDisplay);
    });
  </script>

        <!-- 1. THE FOUNDATION OF ENGLISH SENTENCES -->
        <div class="box highlight">
          <h2>1. The foundation of English sentences 🏗️</h2>
          <p>Every English sentence needs <strong>a subject and a verb</strong>. This is the minimum requirement.</p>
          <ul class="list">
            <li><strong>Subject</strong> = who or what does the action (I, you, he, the cat, Martin…)</li>
            <li><strong>Verb</strong> = the action or state (play, go, is, have…)</li>
            <li>English relies on <strong>fixed word order</strong>, not on changing word endings</li>
          </ul>
          <p>The basic pattern: <span class="formula">Subject + Verb</span></p>
          <ul class="list">
            <li><span class="formula">I play.</span></li>
            <li><span class="formula">She reads.</span></li>
            <li><span class="formula">They go.</span></li>
          </ul>
        </div>

        <!-- 2. SUBJECT + VERB PATTERN -->
        <div class="box">
          <h2>2. Subject + Verb pattern in Present Simple 📐</h2>

          <div style="display:grid; grid-template-columns:1fr 1fr; gap:8px;">

            <div>
              <p><strong>Regular verbs – base form or +s:</strong></p>
              <ul class="list">
                <li><span class="formula">I work.</span></li>
                <li><span class="formula">You play.</span></li>
                <li><span class="formula">He watches.</span> (3rd person singular)</li>
                <li><span class="formula">We study.</span></li>
                <li><span class="formula">She lives.</span> (3rd person singular)</li>
              </ul>
            </div>

            <div>
              <p><strong>Irregular verbs – special forms:</strong></p>
              <ul class="list">
                <li><span class="formula">I am.</span> (be)</li>
                <li><span class="formula">You have.</span></li>
                <li><span class="formula">He is.</span> (be – 3rd person)</li>
                <li><span class="formula">She goes.</span> (go – 3rd person)</li>
                <li><span class="formula">It does.</span> (do – 3rd person)</li>
              </ul>
            </div>

          </div>

          <p><strong>Key principle:</strong> In Present Simple, only 3rd person singular (he, she, it) changes the verb form.</p>
        </div>

        <!-- 3. ADDING OBJECTS AND COMPLEMENTS -->
        <div class="box">
          <h2>3. Expanding the sentence: Subject + Verb + Object 📦</h2>

          <div style="display:grid; grid-template-columns:1fr 1fr; gap:8px;">

            <div>
              <p><strong>Regular verbs with objects:</strong></p>

                <li><span class="formula">He likes pizza.</span></li>

                <li><span class="formula">They clean the house.</span></li>
              </ul>
            </div>

            <div>
              <p><strong>Irregular verbs with objects:</strong></p>
              <ul class="list">
                <li><span class="formula">I have a dog.</span></li>
                <li><span class="formula">You know the answer.</span></li>
                <li><span class="formula">She makes breakfast.</span></li>
                <li><span class="formula">He does homework.</span></li>
                <li><span class="formula">We are students.</span></li>
              </ul>
            </div>

          </div>

          <p><strong>Word order is strict:</strong> Subject → Verb → Object (never change this order in affirmative sentences!)</p>

        <!-- 8. KEY TAKEAWAYS -->
        <div class="box highlight">
          <h2>8. Key takeaways 🎯</h2>

          <p>Remember these fundamental rules for building simple sentences:</p>

          <ul class="list">
            <li><strong>Word order is fixed:</strong> Subject + Verb (+ Object) (+ Preposition + Object)</li>
            <li><strong>Regular verbs:</strong> add -s/-es for 3rd person singular (he, she, it)</li>
            <li><strong>Irregular verbs:</strong> have special forms (be → is/am/are, have → has, go → goes, do → does)</li>
            <li><strong>Prepositions matter:</strong> IN (enclosed), ON (surface), AT (point)</li>
            <li><strong>Don't change word order</strong> – English depends on position, not endings</li>
          </ul>

          <p>Practice building sentences by following this pattern, and soon it will become natural!</p>
